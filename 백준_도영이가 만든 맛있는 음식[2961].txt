// 백준 2961 : 도영이가 만든 맛있는 음식
#include <iostream>
#include <string>
#include <cmath>
#include <vector>

using namespace std;

int N;
vector<pair<int, int>> v;
int mini = 1000000000 + 1;

void cook(int i, int food_S, int food_B) {
	food_S *= v[i].first;
	food_B += v[i].second;

	if (mini > abs(food_S - food_B)) mini = abs(food_S - food_B);

	for (int j = i+1; j < N; j++) {
		cook(j, food_S, food_B);
	}
}

int main() {
	cin >> N;
	for (int i = 0; i < N; i++) {
		int S, B;
		cin >> S >> B;
		v.push_back({ S, B });
	}

	for (int i = 0; i < N; i++) {
		cook(i, 1, 0); // 신맛은 곱이고 쓴맛은 합이다.
	}

	cout << mini << '\n';
}
