// 백준 18312 : 시각
#include <vector>
#include <string>
#include <iostream>

using namespace std;

int main() {
	int n, k;
	cin >> n >> k;
	string str_k = to_string(k);

	int count = 0;
	for (int h = 0; h <= n; h++) {
		for (int m = 0; m <= 59; m++) {
			for (int s = 0; s <= 59; s++) {
				string clock = to_string(h) + to_string(m) + to_string(s);
				
				if (k == 0 && clock.size() != 6) {
					count++;
					continue;
				}
				
				for (int i = 0; i < clock.size(); i++) {
					if (str_k[0] == clock[i]) {
						count++;
						break;
					}
				}
			}
		}
	}
	cout << count << endl;
}
