// 백준 17626 : Four Squares
#include <string>
#include <vector>
#include <iostream>

using namespace std;

int n;
bool chk[50001]; // 사용할 인덱스 : 1~50000

int pow(int k, int n) {
	if (k == 1) return chk[n];

	for (int i = 1; i * i <= n; i++) {
		if (pow(k - 1, n - i * i)) return 1;
	}
	return 0;
}

int main() {
	cin >> n;
	for (int i = 1; i * i <= n; i++) {
		chk[i * i] = true;
	}

	for (int i = 1; i <= 4; i++) {
		if (pow(i, n)) {
			cout << i << '\n';
			break;
		}
	}
	return 0;
}
