// 백준 14500 : 테트로미노 (DFS를 활용한 완전탐색)
#include <vector>
#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

int N, M;
int board[500][500];
int ans = 0;
bool chk[500][500];
int dy[4] = { 0, 0, 1, -1 };
int dx[4] = { 1, -1, 0, 0 };

int DFS(int y, int x, int cnt) {
	if (cnt == 4) return board[y][x];

	int sum = 0;
	for (int i = 0; i < 4; i++) {
		int nxtY = y + dy[i];
		int nxtX = x + dx[i];

		if (0 <= nxtY && nxtY < N && 0 <= nxtX && nxtX < M) {
			if (!chk[nxtY][nxtX]) {
				chk[nxtY][nxtX] = true;
				sum = max(sum, board[y][x] + DFS(nxtY, nxtX, cnt + 1));
				chk[nxtY][nxtX] = false;
			}
		}
	}
	return sum;
}

// DFS탬색으로 처리 불가는한 BLOCK T
int BLOCK_T(int y, int x) {
	int sum = 0;

	if (y - 1 >= 0 && x + 2 < M) {
		sum = max(sum, board[y][x] + board[y][x + 1] + board[y][x + 2] + board[y - 1][x + 1]);
	}

	if (y + 2 < N && x + 1 < M) {
		sum = max(sum, board[y][x] + board[y + 1][x] + board[y + 2][x] + board[y + 1][x + 1]);
	}

	if (y + 1 < N && x + 2 < M) {
		sum = max(sum, board[y][x] + board[y][x + 1] + board[y][x + 2] + board[y + 1][x + 1]);
	}

	if (y + 2 < N && x - 1 >= 0) {
		sum = max(sum, board[y][x] + board[y + 1][x] + board[y + 2][x] + board[y + 1][x - 1]);
	}

	return sum;
}


int main() {
	cin >> N >> M;
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			cin >> board[i][j];
		}
	}

	// 모든 좌표를 시작점으로 DFS탐색
	for (int y = 0; y < N; y++) {
		for (int x = 0; x < M; x++) {
			chk[y][x] = true;
			ans = max(ans, DFS(y, x, 1));
			ans = max(ans, BLOCK_T(y, x));
			chk[y][x] = false;
		}
	}

	// 답안 출력
	cout << ans << '\n';
}
