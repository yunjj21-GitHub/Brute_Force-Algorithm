// 백준 1548 : 부분 삼각 수열
#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int n;
vector<int> v;
vector<pair<int, int>> comb(3);
int ans;

void combination(int cnt, int nxt) {
	if (cnt == 3) {
		if (comb[0].first + comb[1].first > comb[2].first) {
			int size = (comb[2].second - comb[1].second - 1) + 3;
			if (size > ans) {
				ans = size;
			}
		}
		return;
	}

	for (int i = nxt; i < n; i++) {
		comb[cnt].first = v[i]; // 원소 값
		comb[cnt].second = i; // 해당 원소의 v에서의 인덱스 값
		combination(cnt + 1, i + 1);
	}
}

int main() {
	cin >> n;
	v.resize(n);
	for (int i = 0; i < n; i++) {
		cin >> v[i];
	}

	if (n < 3) ans = n;
	else ans = 2;

	sort(v.begin(), v.end()); // 오름차순 정렬
	combination(0, 0);

	cout << ans << '\n';
}
